### 1.2核心代码：育龄期女性是否在配置范围内

```sql
 public bool ShouldShowFertileAgeFemalePopup(OutPatientInfoFacDto patientInfo)
        {
            if (patientInfo.Sex != "1" && patientInfo.Sex != "女")
                return false;
            int Max = 0;
            int Min = 0;
            ArrayList list = conMgr.GetConstantList("ChildbearingAgeFemaleConfig");
            foreach (var item in list)
            {
                var idProperty = item.GetType().GetProperty("ID");
                var nameProperty = item.GetType().GetProperty("Name");
                if (idProperty != null && nameProperty != null)
                {
                    string id = idProperty.GetValue(item, null)?.ToString() ?? "";
                    string nameValue = nameProperty.GetValue(item, null)?.ToString() ?? "";

                    if (int.TryParse(nameValue, out int value))
                    {
                        if (id.Equals("Max", StringComparison.OrdinalIgnoreCase))
                        {
                            Max = value;
                        }
                        else if (id.Equals("Min", StringComparison.OrdinalIgnoreCase))
                        {
                            Min = value;
                        }
                    }
                }
            }

            //  计算年龄 
            int age = DateTime.Now.Year - patientInfo.Birthday.Year;

            // 调整年龄计算（考虑生日是否已过）
            if (patientInfo.Birthday.Date > DateTime.Now.AddYears(-age))
                age--;

            // 需要使用SQL 配置这两个参数
            return age >= Min && age <= Max;
        }
```
