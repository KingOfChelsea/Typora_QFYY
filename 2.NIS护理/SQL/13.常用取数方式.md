```sql
select

```

## 2.查询的数值不纯,to_number报错()

```sql
-- 创建函数后，可以这样使用：
SELECT to_number_safe('123.45') FROM dual;  -- 返回: 123.45
SELECT to_number_safe('abc') FROM dual;     -- 返回: NULL
SELECT to_number_safe('120/80') FROM dual;  -- 返回: NULL
SELECT to_number_safe('1.23E-4') FROM dual; -- 返回: 0.000123
```

## 3.表结构取数语句

```sql
SELECT  T.TABLE_NAME  as 表名称,
       T.COLUMN_NAME  as 列名,
       case
         when T.DATA_TYPE = 'UNDEFINED' then
          'VARCHAR2(106)'
         else
          T.DATA_TYPE || '(' || case
         when nvl(T.DATA_LENGTH, 0) = 0 and nvl(T.DATA_PRECISION, 0) = 0 then
          ''
         when nvl(T.DATA_LENGTH, 0) <> 0 and nvl(T.DATA_PRECISION, 0) = 0 then
          to_char(nvl(T.DATA_LENGTH, 0))
         when nvl(T.DATA_LENGTH, 0) <> 0 and nvl(T.DATA_PRECISION, 0) <> 0 then
          to_char(nvl(T.DATA_LENGTH, 0)) || ',' ||
          to_char(nvl(T.DATA_PRECISION, 0))
       end || ')' end   as 字段类型,
       T.NULLABLE  as 是否为空,
       C.COMMENTS  as 字段名称
  FROM all_TAB_COLUMNS T, all_COL_COMMENTS C
 WHERE T.TABLE_NAME = C.TABLE_NAME
   AND T.COLUMN_NAME = C.COLUMN_NAME
   AND T.TABLE_NAME = upper('mdc_rcd_out_record_item')
 ORDER BY T.COLUMN_ID ASC;
```

## 4.死锁

```sql
SELECT DISTINCT decode(s.inst_id, 1, 'DB1', 2, 'DB2' ,s.inst_id) 数据库服务器,
                decode(s.BLOCKING_SESSION,'',S.USERNAME,'--'||s.USERNAME) 登陆用户名,
                O.OWNER || '.' || O.OBJECT_NAME 锁的对象,
                S.SID,
                decode(S.BLOCKING_SESSION,
                       '',
                       '根锁为此会话',
                       '根锁会话：' || S.BLOCKING_SESSION) 阻塞SID,
                S.BLOCKING_SESSION_STATUS STATUS,
                (CASE
                  WHEN SQL_TEXT IS NULL THEN
                   '(SID:' || S.SID || ')会话 SQL已跑完'
                  ELSE
                   '(SID:' || S.SID || ')会话 正执行SQL：' || SQL_.SQL_TEXT
                END) SQL_TEXT,
                s.CLIENT_INFO,
                s.program,
                LO.REQUEST,
                S.STATE,
                S.EVENT,
                S.MACHINE,
                S.LOGON_TIME,
                S.INST_ID,
                decode(S.WAIT_TIME,-2,'不确定',-1,'<0.1s',0,to_char(S.SECONDS_IN_WAIT)||'s',to_char(S.WAIT_TIME)||'s') AS 等待时间,
                S.BLOCKING_SESSION,
                s.PROGRAM,
                DECODE(S.BLOCKING_SESSION,
                       '',
                       'ALTER SYSTEM KILL SESSION ''' || S.SID || ',' ||
                       S.SERIAL# ||''' immediate ;','') AS norac_KILL,
               DECODE(S.BLOCKING_SESSION,
                       '',
                       'ALTER SYSTEM KILL SESSION ''' || S.SID || ',' ||
                       S.SERIAL# ||',@'||S.inst_id||''' immediate ;','') rac_KILL
  FROM GV$SESSION S
  LEFT JOIN GV$SQL SQL_
    ON SQL_.SQL_ID = S.SQL_ID
   AND SQL_.INST_ID = S.INST_ID
  JOIN GV$LOCKED_OBJECT L
    ON L.SESSION_ID = S.SID
   AND L.INST_ID = S.INST_ID
  JOIN ALL_OBJECTS O
    ON L.OBJECT_ID = O.OBJECT_ID
  JOIN GV$LOCK LO
    ON (LO.BLOCK != 0 OR LO.REQUEST != 0)
 WHERE LO.SID = L.SESSION_ID
   AND LO.SID = S.SID
   AND LO.INST_ID = L.INST_ID
   AND LO.INST_ID = S.INST_ID
 ORDER BY s.inst_id, S.BLOCKING_SESSION,s.SID DESC;

```
