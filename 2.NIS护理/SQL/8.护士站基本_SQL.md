## 1.药品执行档

```sql
SELECT 
    -- 基本信息合并
    e.exec_sqn AS "执行单流水号",
    e.inpatient_no AS "住院流水号",
    e.patient_no AS "住院病历号",
    e.mo_order AS "医嘱流水号",
    
    -- 药品信息合并
    e.drug_code || ' - ' || e.drug_name AS "药品编码名称",
        -- 精确组合医嘱标识（使用树形符号）
    CASE 
        WHEN e.comb_no IS NOT NULL THEN 
            CASE 
                -- 判断组合中药嘱数量是否为1
                WHEN COUNT(*) OVER (PARTITION BY e.comb_no) = 1 THEN ''
                -- 判断是否是组合中的第一条记录
                WHEN ROW_NUMBER() OVER (PARTITION BY e.comb_no ORDER BY e.exec_sqn) = 1 
                THEN '┍ '
                -- 判断是否是组合中的最后一条记录
                WHEN ROW_NUMBER() OVER (PARTITION BY e.comb_no ORDER BY e.exec_sqn) = 
                     COUNT(*) OVER (PARTITION BY e.comb_no)
                THEN '┕ '
                -- 中间记录
                ELSE '│ '
            END || 
            -- 只有当组合中有多个药品时才显示[组合]标签
            CASE WHEN COUNT(*) OVER (PARTITION BY e.comb_no) > 1 THEN '' ELSE '[X]' END
        ELSE ''
    END AS " ",
    
    CASE 
        WHEN e.specs IS NOT NULL THEN e.drug_name || ' (' || e.specs || ')'
        ELSE e.drug_name
    END AS "药品详情",
    e.base_dose || e.dose_unit || ' / ' || e.price_unit AS "剂量规格",
    
    -- 用法用量合并
    e.use_name || ' ' || 
    COALESCE(TO_CHAR(e.dose_once), '') || e.dose_unit || ' ' || 
    e.frequency_name AS "用法用量",
    
    -- 数量和天数合并
    e.qty_tot || e.min_unit || ' × ' || 
    COALESCE(TO_CHAR(e.use_days), '1') || '天' AS "总量",
    
    -- 执行信息合并
    CASE e.exec_flag
        WHEN '0' THEN '待执行'
        WHEN '1' THEN TO_CHAR(e.exec_date, 'YYYY-MM-DD HH24:MI') || 
                     CASE 
                         WHEN e.exec_usercd IS NOT NULL THEN ' 执行人:' || COALESCE(e.exec_usercd, '') || 
                                                           CASE WHEN e.exec_deptcd IS NOT NULL THEN '(' || e.exec_deptcd || ')' ELSE '' END
                         ELSE ''
                     END
        ELSE '未知'
    END AS "执行状态",
    
    -- 配药信息合并
    CASE e.druged_flag
        WHEN '0' THEN '不需发送'
        WHEN '1' THEN '集中发送'
        WHEN '2' THEN '分散发送'
        WHEN '3' THEN TO_CHAR(e.druged_date, 'YYYY-MM-DD HH24:MI') || 
                     CASE 
                         WHEN e.druged_usercd IS NOT NULL THEN ' 配药人:' || e.druged_usercd || 
                                                             CASE WHEN e.druged_deptcd IS NOT NULL THEN '(' || e.druged_deptcd || ')' ELSE '' END
                         ELSE ''
                     END
        ELSE '未知'
    END AS "配药状态",
    
    -- 双签信息合并
    CASE 
        WHEN e.signfirst_username IS NOT NULL AND e.signsecond_username IS NOT NULL THEN
            '双签完成: ' || TO_CHAR(e.signfirst_date, 'YYYY-MM-DD HH24:MI') || ' ' || e.signfirst_username || 
            ' & ' || TO_CHAR(e.signsecond_date, 'YYYY-MM-DD HH24:MI') || ' ' || e.signsecond_username
        WHEN e.signfirst_username IS NOT NULL THEN
            '首签: ' || TO_CHAR(e.signfirst_date, 'YYYY-MM-DD HH24:MI') || ' ' || e.signfirst_username
        WHEN e.need_sign = '1' THEN
            '需双签'
        ELSE
            '无需双签'
    END AS "双签信息",
    
    -- 配液信息合并
    CASE e.compound_flag
        WHEN '1' THEN 
            CASE e.compound_exec
                WHEN '1' THEN '已配液(' || TO_CHAR(e.compound_date, 'YYYY-MM-DD HH24:MI') || ' ' || e.compound_oper || ')'
                ELSE '需配液'
            END
        ELSE '无需配液'
    END AS "配液状态",
    
    -- 医保和费用信息合并
    CASE e.pub_flag
        WHEN '1' THEN '医保'
        WHEN '0' THEN '自费'
        ELSE '未知'
    END || ' | ' ||
    CASE e.is_own_expense
        WHEN 1 THEN '自费项目'
        ELSE '非自费'
    END AS "费用属性",
    
    -- 备注信息合并
    COALESCE(e.mo_note1, '') || 
    CASE 
        WHEN e.mo_note1 IS NOT NULL AND e.mo_note2 IS NOT NULL THEN '; '
        ELSE ''
    END || 
    COALESCE(e.mo_note2, '') AS "备注信息"
    
FROM 
    MET_IPM_EXECDRUG e
WHERE 
    -- 基础条件
    e.valid_flag = '1'  -- 只查询有效记录
    
    -- 患者筛选条件
    AND e.inpatient_no LIKE '%0000501149'  -- 住院流水号
    -- AND e.patient_no = '123456'  -- 住院病历号
    
    -- 执行状态筛选
    AND e.exec_flag IN ('0', '1')  -- 执行状态(0待执行/1已执行)    
    -- 配药状态筛选
    AND e.druged_flag IN ('0', '1', '2', '3')  -- 配药状态
    -- AND e.druged_flag = '3'  -- 只查已配药
    
ORDER BY 
    CASE WHEN e.exec_flag = '0' THEN 0 ELSE 1 END,  -- 待执行的排前面
    e.use_time ASC, 
    e.exec_sqn
```

## 3.查询中心/西药房/中药房/药房是否发药

```sql
SELECT 
    a.trade_name AS "药品商品名称",
    TO_CHAR(a.apply_date, 'YYYY-MM-DD HH24:MI:SS') AS "申请日期",
    TO_CHAR(a.druged_empl) AS "摆药人",  
    TO_CHAR(a.druged_date, 'YYYY-MM-DD HH24:MI:SS') AS "摆药日期",
    CASE 
        WHEN a.apply_state = 0 THEN '申请'
        WHEN a.apply_state = 1 THEN '打印配药'
        WHEN a.apply_state = 2 THEN '核准'
        WHEN a.apply_state = 3 THEN '作废'
        WHEN a.apply_state = 4 THEN '暂不摆药'
        WHEN a.apply_state = 5 THEN '护士统领'
        ELSE '未知状态(' || TO_CHAR(a.apply_state) || ')'  
    END AS "申请状态",
    TO_CHAR(a.cancel_date, 'YYYY-MM-DD HH24:MI:SS') AS "取消摆药日期",
    TO_CHAR(a.cancel_empl) AS "取消摆药员",  
    CASE 
        WHEN a.cancel_empl IS NOT NULL  THEN '是'
        ELSE '否'
    END AS "是否取消摆药"
FROM 
    pha_com_applyout a
JOIN 
    met_ipm_order b ON a.mo_order = b.mo_order
WHERE 
    b.inpatient_no = 'ZY0010000500279' 
    AND a.trade_name LIKE '%氨甲环酸%' -- 药品名称
```

## 6.护士站资料变更 [转入、转出]

```sql
SELECT 
    clinic_no AS "医疗流水号",
    happen_no AS "发生序号",
    -- 变更类型转换
    CASE shift_type
        WHEN 'RI' THEN '转入'
        WHEN 'RO' THEN '转出'
        WHEN 'RB' THEN '转床'
        WHEN 'K' THEN '接诊'
        WHEN 'C' THEN '结算找回'
        WHEN 'B' THEN '住院登记'
        WHEN 'O' THEN '出院'
        WHEN 'OF' THEN '无费退院'
        WHEN 'BP' THEN '结算清单'
        WHEN 'DG' THEN '诊断'
				WHEN 'CNO' THEN '转出'
				WHEN 'CN' THEN '转入'
        ELSE shift_type
    END AS "变更类型",
    old_data_name AS "原资料名称",
    new_data_name AS "新资料名称",
    -- 变更前后信息整合
    CASE 
        WHEN old_data_name IS NOT NULL AND new_data_name IS NOT NULL 
        THEN old_data_name || ' → ' || new_data_name
        WHEN old_data_name IS NOT NULL THEN '原值: ' || old_data_name
        WHEN new_data_name IS NOT NULL THEN '新值: ' || new_data_name
        ELSE NULL
    END AS "变更详情",
    shift_cause AS "变更原因",
    oper_code AS "操作员",
    TO_CHAR(oper_date, 'YYYY-MM-DD HH24:MI:SS') AS "操作时间",
    -- 操作信息整合
    oper_deptcode || '(' || oper_code || ')' AS "操作科室(操作员)"
FROM com_shiftdata
WHERE clinic_no LIKE '%501232%'
ORDER BY oper_date DESC, happen_no DESC;
```

## 7.医嘱表 [可用于查询皮试]

```sql
SELECT 

    e.hypo_combo || '|' || 
    CASE WHEN e.need_hypo_flag = '1' THEN '需皮试' ELSE '无需' END || '|' ||
    CASE e.hypo_result WHEN '0' THEN '阴性' WHEN '1' THEN '阳性' ELSE '未做' END AS 皮试摘要,
    e.inpatient_no AS 住院流水号,
    e.mo_order AS 医嘱流水号,
    e.combo_no AS 组合号,
    
    -- 医嘱基本信息
    CASE 
        WHEN e.combo_no IS NOT NULL THEN 
            CASE 
                WHEN COUNT(*) OVER (PARTITION BY e.combo_no) = 1 THEN ' ╳ '
                WHEN ROW_NUMBER() OVER (PARTITION BY e.combo_no ORDER BY e.combo_seq) = 1 THEN ' ┏'
                WHEN ROW_NUMBER() OVER (PARTITION BY e.combo_no ORDER BY e.combo_seq) = 
                     COUNT(*) OVER (PARTITION BY e.combo_no) THEN ' ┗'
                ELSE ' ┣'
            END
        ELSE ''
    END AS " ",
    
    CASE WHEN e.emc_flag = '1' THEN '【加急】' END || e.order_name AS 医嘱名称,
    CASE e.item_type 
        WHEN '1' THEN '药品' 
        WHEN '2' THEN '诊疗' 
        ELSE e.item_type 
    END AS 项目类型,
    

    
    -- 医嘱状态信息
    CASE e.mo_state
        WHEN '10' THEN '待审核'
        WHEN '15' THEN '保存未提交'
        WHEN '18' THEN '驳回'
        WHEN '20' THEN '已提交'
        WHEN '30' THEN '已接收'
        WHEN '40' THEN '已执行'
        WHEN '50' THEN '已完成'
        WHEN '90' THEN '停止作废'
        ELSE e.mo_state
    END AS 医嘱状态,
    
    -- 费用信息
    '￥' || e.cost_ref AS 参考金额,
    CASE e.charge_flag 
        WHEN '1' THEN '已计费' 
        ELSE '未计费' 
    END AS 计费状态,
    
    -- 时间信息整合
    e.begin_date AS 频次推荐执行,
    e.end_date AS 停止时间,
    e.mo_date AS 开立时间,
    e.confirm_date AS 接收时间,
    e.exec_date AS 执行时间,
    e.finish_date AS 完成时间,
    
    -- 人员信息整合
    e.mo_doc AS 开立医生,
    e.mo_dept AS 开立科室,
    e.confirm_oper AS 接收护士,
    e.exec_oper AS 执行人,
    
    -- 药品剂量信息（如为药品医嘱）
    CASE WHEN e.item_type = '1' THEN
        e.dose_once || e.dose_unit || ' ' || e.frequency_code || ' ' || e.usage_code
    END AS 用药方案,
    
    CASE WHEN e.item_type = '1' THEN
        e.tot_qty || e.tot_unit || ' × ' || e.days || '天'
    END AS 总量信息,
    
    -- 特殊医嘱类型
    CASE e.ext_data8
        WHEN '1' THEN '输血'
        WHEN '2' THEN '手术'
        WHEN '3' THEN '会诊'
        WHEN '4' THEN '病理'
        WHEN '5' THEN '多科室会诊'
        ELSE '普通医嘱'
    END AS 特殊医嘱类型,
    
    e.ext_data9 AS 申请单号,
    
    -- 抗生素使用类型
    CASE e.antibiotic_type
        WHEN '1' THEN '预防性'
        WHEN '2' THEN '治疗性'
        ELSE '未知'
    END AS 抗生素类型,
    
    -- 特殊标识整合
    CASE WHEN e.emc_flag = '1' THEN '加急' END AS 加急标识,
    CASE WHEN e.main_flag = '1' THEN '主要医嘱' END AS 主要标识,
    CASE WHEN e.nurse_order_flag = '1' THEN '护嘱' END AS 护嘱标识,
    
    -- 执行信息
    e.exec_times AS 执行时间点,
    e.dropping_speed || e.dropping_speed_unit AS 滴速,
    
    -- 驳回信息
    CASE WHEN e.mo_state = '18' THEN
        e.refuse_doc || '于' || TO_CHAR(e.refuse_date, 'YYYY-MM-DD HH24:MI') || '驳回:' || e.refuse_reason
    END AS 驳回信息,
    
    -- 停止信息
    CASE WHEN e.mo_state = '90' THEN
        e.dc_doc || '于' || TO_CHAR(e.dc_date, 'YYYY-MM-DD HH24:MI') || '停止:' || e.dc_reason
    END AS 停止信息
    
FROM met_ordi_order e
WHERE 1=1
AND e.inpatient_no LIKE '%ZY0010000501810%'  
ORDER BY 
    COALESCE(e.combo_no, e.mo_order),
    e.combo_no DESC,
    e.begin_date DESC;
```
