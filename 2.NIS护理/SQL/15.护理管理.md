### 1.5 发布任务后评分

```sql
SELECT (SELECT a.node_name 
        FROM nis_qc_standard_node a 
        WHERE a.node_id = n.parent_id) AS 标准跟节点,
       n.node_name AS 标准名称,
       p.Plan_Name AS 任务名称,
       p.start_date AS 开始时间,
       p.end_date AS 结束时间,
       fun_get_empl_name(p.create_code) AS 创建人,
       r.score_result as 评分结果,
       r.score_full as 满分状态,
       r.score_state as 评分状态,
       fun_get_dept_name(r.dept_code) as 评估科室,
       fun_get_empl_name(r.oper_code) as 评分人,
       -- 新增得分率字段
       CASE 
           WHEN r.score_full > 0 
           THEN TO_CHAR(ROUND(r.score_result / r.score_full * 100, 2), 'FM999990.00') || '%'
           ELSE '0.00%'
       END AS 得分率
       -- r.*
FROM nis_qc_plan p
LEFT JOIN nis_qc_standard_node n ON p.first_level_id = n.node_id
LEFT JOIN nis_qc_plan_score_record r ON p.id = r.plan_id
WHERE p.start_date = TO_DATE('2026-03-01', 'YYYY-MM-DD') -- 开始时间
  AND p.end_date = TO_DATE('2026-03-31', 'YYYY-MM-DD') -- 结束时间
  AND p.create_code = 'CH1844' -- 计划创建人
  AND p.plan_status = 1 -- 发布状态
  AND p.valid_flag = 1 -- 有效值
  AND r.score_state = 1;  -- 控制整个任务是否全部按照次数全部完成，像之前的旧数据就需要全部关闭
```

- score_state `控制整份任务是否完成，是否全部完成需要`

### 1.6 按照标准名称和科室分组-计算总得分率

```sql
-- ================ 合并报表-按照标准名称和科室分组，计算总得分率 ====================
WITH score_data AS (
    SELECT 
        (SELECT a.node_name 
         FROM nis_qc_standard_node a 
         WHERE a.node_id = n.parent_id) AS 标准根节点,
        n.node_name AS 标准名称,
        p.Plan_Name AS 任务名称,
        p.start_date AS 开始时间,
        p.end_date AS 结束时间,
        fun_get_empl_name(p.create_code) AS 创建人,
        r.score_result AS 评分结果,
        r.score_full AS 满分状态,
        r.score_state AS 评分状态,
        fun_get_dept_name(r.dept_code) AS 评估科室,
        fun_get_empl_name(r.oper_code) AS 评分人,
        r.dept_code AS 科室编码
    FROM nis_qc_plan p
    LEFT JOIN nis_qc_standard_node n ON p.first_level_id = n.node_id
    LEFT JOIN nis_qc_plan_score_record r ON p.id = r.plan_id
    WHERE p.start_date = TO_DATE('2026-03-01', 'YYYY-MM-DD')
      AND p.end_date = TO_DATE('2026-03-31', 'YYYY-MM-DD')
      AND p.create_code = 'CH1844'
      AND p.plan_status = 1
      AND p.valid_flag = 1
      -- AND r.score_state = 1 
)
-- =============== 主查询：按照标准名称和科室分组，计算总得分率 ====================
SELECT 
    d.标准根节点,
    d.标准名称,
    d.评估科室,
-- ======================== 计算总得分率 =========================
    CASE 
        WHEN SUM(d.满分状态) > 0 
        THEN TO_CHAR(ROUND(SUM(d.评分结果) / SUM(d.满分状态) * 100, 2), 'FM999990.00') || '%'
        ELSE '0.00%'
    END AS 总得分率,
-- ======================== 统计信息 =========================
    COUNT(DISTINCT d.任务名称) AS 任务数量, -- PLAN计划
    COUNT(*) AS 评分记录数,-- 一个PLAN评了几次
    MIN(d.开始时间) AS 最早开始时间,
    MAX(d.结束时间) AS 最晚结束时间
FROM score_data d
GROUP BY 
    d.标准根节点,
    d.标准名称,
    d.评估科室
ORDER BY 
    d.标准根节点,
    d.标准名称,
    d.评估科室;
```

```sql
WITH base_data AS (
    SELECT 
        (SELECT nn.node_name FROM nis_qc_standard_node nn WHERE nn.node_id = n.parent_id) AS 标准根节点,
        n.Node_Name AS 一级节点名称,
        m.deduct_standard AS 标准名称,
        r.record_id,
        fun_get_dept_name(r.dept_code) as dept_code ,
        r.oper_code,
        a.check_num
    FROM nis_qc_plan_score_record r 
    LEFT JOIN nis_qc_plan_score_detail a ON a.score_record_id = r.record_id
    JOIN nis_qc_standard_item m ON a.standard_item_id = m.id
    LEFT JOIN nis_qc_standard_node n ON m.FIRST_LEVEL_NODE_ID = n.node_id
    LEFT JOIN nis_qc_plan p ON p.id = a.plan_id
    WHERE 
        p.start_date = TO_DATE('2026-03-01', 'YYYY-MM-DD')
        AND p.end_date = TO_DATE('2026-03-31', 'YYYY-MM-DD')
        AND p.create_code = 'CH1844'
        AND p.plan_status = 1
        AND p.valid_flag = 1
        AND a.check_num > 0
        --AND n.Node_Name = '护理管理质量评价标准'
)
-- 主查询：获取唯一值组合及统计
SELECT 
    标准根节点,
    一级节点名称,
    标准名称 AS "标准名称(明细规则)",
    COUNT(DISTINCT record_id) AS 发生频次,
    SUM(check_num) AS 总检查例数,
    COUNT(DISTINCT dept_code) AS 涉及科室数,
    LISTAGG(DISTINCT dept_code, ', ') WITHIN GROUP (ORDER BY dept_code) AS 科室列表
FROM base_data
GROUP BY 标准根节点, 一级节点名称, 标准名称
ORDER BY 发生频次 DESC, 标准根节点, 一级节点名称, 标准名称;
```

### 2.5 查看护士排班

```sql
-- 1. 查询当前日期所在月份的排班（整个月）
SELECT 
    a.empl_id || fun_get_empl_name(a.empl_id) as 护士信息,
    a.Class_Abbr as 班次缩写,
    a.shift_data as 班次日期
FROM NIS_SCH_SHIFT a 
WHERE a.empl_id = 'CH9073'
    AND a.is_availability = '1'
    -- shift_data是varchar2类型，需要转为日期进行比较
    AND TO_DATE(a.shift_data, 'YYYY-MM-DD') >= TRUNC(SYSDATE, 'MM')  -- 当月第一天
    AND TO_DATE(a.shift_data, 'YYYY-MM-DD') < ADD_MONTHS(TRUNC(SYSDATE, 'MM'), 1)  -- 下个月第一天
ORDER BY a.shift_data DESC;

-- 2. 查询当前日期所在周的排班（整个周，按周一到周日）
SELECT 
    a.empl_id || fun_get_empl_name(a.empl_id) as 护士信息,
    a.Class_Abbr as 班次缩写,
    a.shift_data as 班次日期
FROM NIS_SCH_SHIFT a 
WHERE a.empl_id = 'CH9073' -- 工号
    AND a.is_availability = '1'
    -- 使用TRUNC(SYSDATE, 'IW')获取本周第一天（周一是第一天）
    AND TO_DATE(a.shift_data, 'YYYY-MM-DD') >= TRUNC(SYSDATE, 'IW')  -- 本周第一天
    AND TO_DATE(a.shift_data, 'YYYY-MM-DD') < TRUNC(SYSDATE, 'IW') + 7  -- 下周第一天
ORDER BY a.shift_data DESC;
```

### 4.1 三级质控数据

```sql
WITH plan_filter AS (
    SELECT id, start_date, end_date, create_code, plan_status, valid_flag
    FROM nis_qc_plan p
    WHERE p.start_date = DATE '2026-03-01'
        AND p.end_date = DATE '2026-03-31'
        AND p.create_code = 'CH1844'
        AND p.plan_status = 1
        AND p.valid_flag = 1
)
SELECT 
    root_node.node_name AS 根节点,
    first_level.node_name AS 一级节点名称,
    second_level.node_name AS 二级节点名称,
    thrid_level.node_name AS 三级节点名称,
    m.deduct_standard AS 标准名称,
    LISTAGG(DISTINCT fun_get_dept_name(r.dept_code), ', ') 
        WITHIN GROUP (ORDER BY fun_get_dept_name(r.dept_code)) AS 所有科室
FROM plan_filter p
LEFT JOIN nis_qc_plan_score_detail a ON a.plan_id = p.id
LEFT JOIN nis_qc_plan_score_record r ON a.score_record_id = r.record_id
LEFT JOIN nis_qc_standard_item m ON a.standard_item_id = m.id
-- 连接到当前节点的父节点
JOIN nis_qc_standard_node n ON m.parent_id = n.node_id
-- 三级节点
JOIN nis_qc_standard_node thrid_level ON n.node_id = thrid_level.node_id
    AND thrid_level.node_level = '4' -- 三级节点
-- 二级节点
JOIN nis_qc_standard_node second_level ON thrid_level.parent_id = second_level.node_id
    AND second_level.node_level = '3' -- 二级节点
-- 一级节点
JOIN nis_qc_standard_node first_level ON second_level.parent_id  = first_level.node_id 
    AND first_level.node_level = '2'  -- 一级节点
-- 根节点
JOIN nis_qc_standard_node root_node ON first_level.parent_id = root_node.node_id
    AND root_node.node_level = '1'  -- 根节点
WHERE a.check_num > 0
GROUP BY 
    root_node.node_name,
    first_level.node_name,
    second_level.node_name,
    thrid_level.node_name,
    m.deduct_standard
ORDER BY 
    root_node.node_name,
    first_level.node_name,
    second_level.node_name,
    thrid_level.node_name;
```

### 5.1 三级指标获取条目数

```sql
WITH plan_filter AS (
    SELECT id, start_date, end_date, create_code, plan_status, valid_flag
    FROM nis_qc_plan p
    WHERE 1=1 
		     and p.start_date = DATE '2026-03-01'
        AND p.end_date = DATE '2026-03-31'
        AND p.create_code = 'CH1844'
        AND p.plan_status = 1
        AND p.valid_flag = 1
),
base_data AS (
    SELECT 
        -- 三级节点 
        thrid_level.node_name AS 三级节点名称,
        -- 一级节点
        first_level.node_name AS 一级节点名称,
        -- 二级节点
        second_level.node_name AS 二级节点名称,
        -- 根节点
        root_node.node_name AS 当前节点名称,
        -- 标准名称
        m.deduct_standard AS 标准名称,
        -- 科室信息
        fun_get_dept_name(r.dept_code) AS dept_name,
        r.dept_code,
        -- 备注信息
        r.score_comment, -- 备注
        -- 其他信息
        r.record_id,
        r.oper_code,
        a.check_num,
        -- 排序字段
        root_node.sort_num AS root_sort,
        first_level.sort_num AS first_sort,
        second_level.sort_num AS second_sort,
        thrid_level.sort_num AS third_sort
    FROM plan_filter p
    LEFT JOIN nis_qc_plan_score_detail a ON a.plan_id = p.id
    LEFT JOIN nis_qc_plan_score_record r ON a.score_record_id = r.record_id
    LEFT JOIN nis_qc_standard_item m ON a.standard_item_id = m.id
    -- 连接到标准节点的父节点
    JOIN nis_qc_standard_node n ON m.parent_id = n.node_id
    -- 三级节点（当前节点）
    JOIN nis_qc_standard_node thrid_level ON n.node_id = thrid_level.node_id
        AND thrid_level.node_level = '4'
    -- 二级节点
    JOIN nis_qc_standard_node second_level ON thrid_level.parent_id = second_level.node_id
        AND second_level.node_level = '3'
    -- 一级节点
    JOIN nis_qc_standard_node first_level ON second_level.parent_id = first_level.node_id 
        AND first_level.node_level = '2'
    -- 根节点
    JOIN nis_qc_standard_node root_node ON first_level.parent_id = root_node.node_id
        AND root_node.node_level = '1'
    WHERE a.check_num >= 0
        AND a.plan_id IS NOT NULL
        AND a.standard_item_id IS NOT NULL
)
SELECT 
    当前节点名称,
    一级节点名称,
    二级节点名称,
    三级节点名称,
    标准名称,
    -- 1. 科室和备注组合显示，格式："科室名称：备注"
    LISTAGG(
        dept_name || '：' || score_comment, 
        '，'
    ) WITHIN GROUP (ORDER BY dept_name) AS 科室及备注,
    -- 2. 所有科室（去重）
LISTAGG(dept_name || '：' || NVL(score_comment, '无备注'), ' ; ' || CHR(13) || CHR(10)) 
WITHIN GROUP (ORDER BY dept_name) AS 科室及备注,
    -- 也可以分开显示
    LISTAGG(DISTINCT dept_name, ', ') 
        WITHIN GROUP (ORDER BY dept_name) AS 所有科室,
    LISTAGG(DISTINCT score_comment, '；') 
        WITHIN GROUP (ORDER BY score_comment) AS 所有备注,
    -- 4. 涉及科室数
    COUNT(DISTINCT dept_code) AS 涉及科室数,
    -- 5. 检查记录数
    COUNT(DISTINCT record_id) AS 检查记录数,
    -- 6. 检查总次数
    SUM(check_num) AS 检查总次数
FROM base_data
GROUP BY 
    当前节点名称,
    一级节点名称,
    二级节点名称,
    三级节点名称,
    标准名称
ORDER BY 
    三级节点名称,
    二级节点名称,
    一级节点名称,
    当前节点名称,
    标准名称;
```

```sql
            ELSE 0
        END AS 总得分率,
        -- 统计信息
        SUM(d.评分结果) AS 总分,
        SUM(d.满分状态) AS 满分,
        COUNT(DISTINCT d.任务名称) AS 任务数量,
        COUNT(*) AS 评分记录数,
        MIN(d.开始时间) AS 最早开始时间,
        MAX(d.结束时间) AS 最晚结束时间
    FROM score_data d
    GROUP BY 
        d.标准根节点,
        d.标准名称,
        d.评估科室
)
-- 使用PIVOT进行转置，将科室转为列
SELECT *
FROM (
    SELECT 
        标准根节点,
        标准名称,
        评估科室,
        总得分率,
        总分,
        满分
    FROM dept_scores
    -- 添加条件，排除得分率为0或NULL的记录
    WHERE 总得分率 > 0
		AND 总分 is NOT NULL 
)
PIVOT (
    MAX(总得分率) AS 得分率,
    MAX(总分) AS 总分,
    MAX(满分) AS 满分
    FOR 评估科室 IN (
        '产房/产前病区' AS 产房产前病区,
        '产科(爱婴区)护理组' AS 产科爱婴区护理组,
        '儿科病区' AS 儿科病区,
        '肺病科(呼吸内科)病区' AS 肺病科病区,
        '妇产科贵宾区' AS 妇产科贵宾区,
        '骨伤科(骨二科)病区' AS 骨二科病区,
        '骨伤科(骨一科)病区' AS 骨一科病区,
        '护老中心护理组' AS 护老中心护理组,
        '康复科一病区' AS 康复科一病区,
        '脑病科(神经内科)病区' AS 脑病科病区,
        '内分泌科病区' AS 内分泌科病区,
        '脾胃病科(消化内科)病区' AS 脾胃病科病区,
        '神志病科(心理科)病区' AS 神志病科病区,
        '肾病科(肾内科)病区' AS 肾病科病区,

        '外五病区(耳鼻喉头颈外科、眼科、肛肠科)' AS 外五病区,
        '外一病区(神经外科、泌尿外科)' AS 外一病区,
        '心病科(心血管内科)病区' AS 心病科病区,

    )

ORDER BY 标准根节点, 标准名称;
```
