# 各类评估单高危患者取数

## 1.跌倒高危患者

### 1.1 跌倒高危患者明细

```sql
WITH 
-- 1. 生成从2025-09-26到当前日期的所有日期
date_range AS (
  SELECT 
    TO_DATE('2026-01-18', 'YYYY-MM-DD') + LEVEL - 1 as query_date
  FROM DUAL
  CONNECT BY TO_DATE('2026-01-18', 'YYYY-MM-DD') + LEVEL - 1 <= TRUNC(SYSDATE)
),

-- 2. 对每个查询日期获取在院患者列表
inpatients_by_date AS (
  SELECT 
    d.query_date,
    a.name,
    a.bed_no,
    a.inpatient_no,
    a.nurse_cell_code
  FROM date_range d
  CROSS JOIN fin_ipr_inmaininfo a 
  WHERE 1=1
    AND TRUNC(a.In_Date) <= d.query_date
    AND (
      a.Out_Date IS NULL 
      OR a.Out_Date < DATE '1000-01-01' --过滤远古日期
      OR TRUNC(a.Out_Date) > d.query_date
    )
    AND a.inpatient_no IS NOT NULL
    AND a.nurse_cell_code IS NOT NULL
    AND a.nurse_cell_code <> 'cehsz09'
    AND a.bed_no is not  null 
		and a.nurse_cell_code = '45700'
),

-- 3. 对每个查询日期获取患者最新转科记录
latest_shift_by_date AS (
  SELECT 
    d.query_date,
    cs.clinic_no,
    cs.new_data_code,
    ROW_NUMBER() OVER (
      PARTITION BY d.query_date, cs.clinic_no 
      ORDER BY cs.happen_no DESC
    ) AS shift_rn
  FROM date_range d
  LEFT JOIN com_shiftdata cs ON TRUNC(cs.oper_date) <= d.query_date
  WHERE cs.shift_type = 'CNO'
),

-- 4. 对每个查询日期获取跌倒风险评估记录
fall_assessments_by_date AS (
  SELECT
    d.query_date,
    nr.inpatient_code as patient_id,
    TRUNC(ac.create_date) AS assess_date,
    TO_NUMBER(REGEXP_SUBSTR(nsi.item_content, '\d+')) as fall_score,
    nr.dept_code as assess_dept_code,
    nsi.common_tree_id,
    CASE 
      -- 成人跌倒高风险
      WHEN nsi.common_tree_id IN (
        '646cfc10-f240-423e-ba4f-75a9da7c182f',
        '26b1aa27-728e-4197-9d44-52a0d6515fa1',
        'c16ca477-a36e-4d18-a412-1a640ce1ae2a'
      ) 
      AND TO_NUMBER(REGEXP_SUBSTR(nsi.item_content, '\d+')) > 13
      THEN '高风险'
      
      -- 儿童跌倒高风险
      WHEN nsi.common_tree_id IN (
        '138d3320-65f3-4c2e-a077-4dfcffd5ac81'
      ) 
      AND TO_NUMBER(REGEXP_SUBSTR(nsi.item_content, '\d+')) > 11
      THEN '高风险'
      ELSE '非高风险'
    END as risk_level,
    ROW_NUMBER() OVER (
      PARTITION BY d.query_date, nr.inpatient_code
      ORDER BY ac.create_date DESC
    ) as assess_rn
  FROM date_range d
  LEFT JOIN Nis_Nas_Record_Item nsi ON 1=1
  LEFT JOIN Nis_Nas_Record nr ON nsi.record_id = nr.id
  LEFT JOIN Nis_Nas_Activity ac ON nr.activity_id = ac.id
  WHERE nsi.common_tree_id IN (
    '646cfc10-f240-423e-ba4f-75a9da7c182f',
    '138d3320-65f3-4c2e-a077-4dfcffd5ac81',
    '26b1aa27-728e-4197-9d44-52a0d6515fa1',
    'c16ca477-a36e-4d18-a412-1a640ce1ae2a'
  )
  AND nsi.valid_state = 1
  AND TRUNC(ac.create_date) <= d.query_date
),

-- 5. 每个患者在每个查询日期的最新跌倒风险评估
latest_fall_assessments_by_date AS (
  SELECT 
    query_date,
    patient_id,
    assess_date,
    fall_score,
    assess_dept_code,
    common_tree_id,
    risk_level
  FROM fall_assessments_by_date
  WHERE assess_rn = 1
)

  SELECT 
    d.query_date as record_date,
    i.inpatient_no as inpatient_no,
    SUBSTR(i.BED_NO, LENGTH(i.nurse_cell_code) + 1) as bed_no,
    i.name as inpatient_name,
    -- 确定当前科室
    COALESCE(ls.new_data_code, i.nurse_cell_code) as dept_code, --当前护理单元编码 
    fun_get_dept_name(COALESCE(ls.new_data_code, i.nurse_cell_code)) as dept_name, -- 当前护理单元名称
    lfa.assess_date as assess_date, --评估日期
    lfa.fall_score as fall_score,--跌倒分数
    lfa.common_tree_id as common_tree_id, -- 评估树ID
    lfa.risk_level as risk_level, -- 风险等级
    lfa.assess_dept_code as assess_dept_code,-- 评估科室编码
    fun_get_dept_name(lfa.assess_dept_code) as assess_dept_name -- 评估科室名称
    -- 添加标记是否在院
    --'是' as 是否在院
  FROM date_range d
  LEFT JOIN inpatients_by_date i ON d.query_date = i.query_date
  LEFT JOIN latest_shift_by_date ls ON d.query_date = ls.query_date 
    AND i.inpatient_no = ls.clinic_no 
    AND ls.shift_rn = 1 --个人评估最新排序
  LEFT JOIN latest_fall_assessments_by_date lfa ON d.query_date = lfa.query_date 
    AND i.inpatient_no = lfa.patient_id
		--AND risk_level = '高风险'

	order by record_date desc, to_number(BED_NO) desc	
```

### 1.2 跌倒高危患者按照病区统计

WITH 
-- 1. 生成从2025-09-26到当前日期的所有日期
date_range AS (

),

-- 3. 对每个查询日期获取患者最新转科记录
latest_shift_by_date AS (
  SELECT 
    d.query_date,
    cs.clinic_no,
    cs.new_data_code,
    ROW_NUMBER() OVER (
      PARTITION BY d.query_date, cs.clinic_no 
      ORDER BY cs.happen_no DESC
    ) AS shift_rn
  FROM date_range d
  LEFT JOIN com_shiftdata cs ON TRUNC(cs.oper_date) <= d.query_date
  WHERE cs.shift_type = 'CNO'
),

-- 4. 对每个查询日期获取跌倒风险评估记录
fall_assessments_by_date AS (
  SELECT
    d.query_date,
    nr.inpatient_code as patient_id,
    TRUNC(ac.create_date) AS assess_date,
    TO_NUMBER(REGEXP_SUBSTR(nsi.item_content, '\d+')) as fall_score,
    nr.dept_code as assess_dept_code,
    nsi.common_tree_id,
    CASE 
      -- 成人跌倒高风险

        '646cfc10-f240-423e-ba4f-75a9da7c182f',
        '26b1aa27-728e-4197-9d44-52a0d6515fa1',
        'c16ca477-a36e-4d18-a412-1a640ce1ae2a'
      ) 
      AND TO_NUMBER(REGEXP_SUBSTR(nsi.item_content, '\d+')) > 13
      THEN '高风险'
      
      -- 儿童跌倒高风险
      WHEN nsi.common_tree_id IN (
        '138d3320-65f3-4c2e-a077-4dfcffd5ac81'
      ) 
      AND TO_NUMBER(REGEXP_SUBSTR(nsi.item_content, '\d+')) > 11
      THEN '高风险'
      
      ELSE '非高风险'
    END as risk_level,
    ROW_NUMBER() OVER (
      PARTITION BY d.query_date, nr.inpatient_code
      ORDER BY ac.create_date DESC
    ) as assess_rn
  FROM date_range d
  LEFT JOIN Nis_Nas_Record_Item nsi ON 1=1
  LEFT JOIN Nis_Nas_Record nr ON nsi.record_id = nr.id
  LEFT JOIN Nis_Nas_Activity ac ON nr.activity_id = ac.id
  WHERE nsi.common_tree_id IN (
    '646cfc10-f240-423e-ba4f-75a9da7c182f',
    '138d3320-65f3-4c2e-a077-4dfcffd5ac81',
    '26b1aa27-728e-4197-9d44-52a0d6515fa1',
    'c16ca477-a36e-4d18-a412-1a640ce1ae2a'

-- 7. 每日科室统计（包含高风险和非高风险）
daily_dept_stats AS (
  SELECT 
    record_date,
    dept_name as dept_name,
    --fun_get_dept_name(当前护理单元编码) as 科室名称,
		dept_code as dept_code,
   -- COUNT(*) as 在院总人数,
    SUM(CASE WHEN risk_level = '高风险' THEN 1 ELSE 0 END) as risk_item_count
    --SUM(CASE WHEN 风险等级 IS NULL THEN 1 ELSE 0 END) as 未评估人数,
    --COUNT(DISTINCT 评估单类型) as 评估类型数量
  FROM all_patients_by_date
  GROUP BY record_date, dept_code,dept_name
)

-- 方案A: 按日期显示统计结果
SELECT 
  record_date,
	dept_name,
	dept_code,
	risk_item_count
FROM daily_dept_stats
ORDER BY record_date DESC, risk_item_count DESC, dept_code;
