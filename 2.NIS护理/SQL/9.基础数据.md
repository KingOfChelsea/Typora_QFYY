## 1.基础人员信息

```sql
SELECT 
    e.empl_id AS "员工ID",
    e.empl_name AS "姓名",
    CASE e.sex_code
        WHEN '1' THEN '男'
        WHEN '2' THEN '女'
        ELSE '未知'
    END AS "性别",
    e.dept_id AS "科室ID",
    e.posi_code AS "职务",
    CASE e.valid_state
        WHEN '1' THEN '在职'
        WHEN '0' THEN '停用'
        ELSE '其他'
    END AS "状态",
    e.tel AS "联系电话",
    TO_CHAR(e.create_date, 'YYYY-MM-DD') AS "入职日期"
FROM 
    dawn_org_empl e
WHERE 1=1
    and e.valid_state = '1'  -- 只查询在职员工
    -- AND e.dept_id = 'DEP001'  -- 可按科室筛选
ORDER BY 
    e.dept_id,
    e.empl_name
```
