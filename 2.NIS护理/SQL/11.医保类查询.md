# 医保类查询

## 1.发票结算

```sql
-- 查询指定住院号的所有结算记录
SELECT 
    (select a.name || '(' || a.patient_no || ')' from fin_ipr_inmaininfo a where  bh.inpatient_no = a.inpatient_no ) AS 住院号,
	pui.pact_name AS 合同单位名称,
    bh.invoice_no AS 发票号,
    bh.balance_no AS 结算序号,
    bh.balance_type AS 结算类型,
    bh.tot_cost AS 总费用,
    bh.own_cost AS 自费金额,
    bh.pay_cost AS 自付金额,
    bh.pub_cost AS 公费金额,
    bh.cash_pay AS 现金支付,
    bh.account_pay AS 账户支付,
    bh.begin_date AS 起始日期,
    bh.end_date AS 终止日期,
    bh.balance_date AS 结算时间
FROM fin_ipb_balancehead bh
LEFT JOIN fin_com_pactunitinfo pui 
    ON bh.pact_code = pui.pact_code 
    AND bh.paykind_code = pui.paykind_code
WHERE 1=1 
AND bh.inpatient_no = 'ZY0130000371980'
ORDER BY bh.balance_date DESC;
```

## 2.广东省直市直统一医保信息主表

```sql
--广东省直市直统一医保信息主表 
SELECT 
    gd.inpatient_no AS 住院流水号,
    gd.reg_no AS 就医登记号,
    gd.name AS 姓名,
    gd.idenno AS 身份证号,
    gd.sex_code AS 性别,
    CASE gd.empl_type 
        WHEN '1' THEN '在职' 
        WHEN '2' THEN '退休' 
        ELSE gd.empl_type 
    END AS 人员类别,
    gd.dept_name AS 科室名称,
    gd.bed_no AS 床号,
    gd.in_date AS 入院日期,
    gd.out_date AS 出院日期,
    gd.in_diagnosename AS 入院诊断,
    gd.out_diagnosename AS 出院诊断,
    gd.tot_cost AS 总费用,
    gd.pub_cost AS 社保支付,
    gd.own_cost AS 个人自负,
    gd.item_paycost AS 项目自付,
    gd.item_paycost2 AS 自费项目金额,
    gd.base_cost AS 起付金额,
    gd.balance_date AS 结算日期,
    CASE gd.balance_state 
        WHEN '1' THEN '已结算' 
        ELSE '未结算' 
    END AS 结算状态
FROM fin_ipr_siinmaininfo_gd gd
WHERE 1=1 
AND gd.patient_no like '%ZY0130000371980%'
ORDER BY gd.balance_date DESC NULLS LAST;
```
