### 3.2 查看表空间

`合并表空间使用+最大可扩展统计，计算最大可分配实际使用率`

```sql
-- 最终修复版：合并表空间使用+最大可扩展统计，计算最大可分配实际使用率
SELECT
    SYSDATE AS 时间,
    a.tablespace_name AS 表空间,
    ROUND(a.total_avail / POWER(1024, 3), 2) AS 总共_GB,
    ROUND(a.total_used / POWER(1024, 3), 2) AS 已经使用_GB,
    ROUND(a.total_free / POWER(1024, 3), 2) AS 空闲_GB,
    ROUND((a.total_used * 100) / a.total_avail, 2) AS "已分配占用率_%",
    ROUND((a.total_free * 100) / a.total_avail, 2) AS "已分配空闲率_%",
    COALESCE(b.file_count, 0) AS 文件数量,
    ROUND(COALESCE(b.total_allocated / POWER(1024, 3), 0), 2) AS 实际分配总大小_GB,
    ROUND(COALESCE(b.max_extend / POWER(1024, 3), 0), 2) AS 最大可扩展总大小_GB,
    -- 核心需求：最大可分配的实际使用率（已使用/最大可扩展）
    ROUND(CASE 
        WHEN b.max_extend = 0 THEN 0 
        ELSE (a.total_used * 100) / b.max_extend 
    END, 2) AS "最大可分配实际使用率_%",
    -- 分配率（当前分配/最大可扩展）
    ROUND(CASE 
        WHEN b.max_extend = 0 THEN 0 
        ELSE (b.total_allocated * 100) / b.max_extend 
    END, 2) AS "分配率_%"
FROM (
    SELECT
        ts_avail.tablespace_name,
        ts_avail.bytes AS total_avail,
        ts_used.bytes AS total_used,
        ts_free.bytes AS total_free
    FROM SYS.SM$TS_AVAIL ts_avail
    INNER JOIN SYS.SM$TS_USED ts_used 
        ON ts_avail.tablespace_name = ts_used.tablespace_name
    INNER JOIN SYS.SM$TS_FREE ts_free 
        ON ts_avail.tablespace_name = ts_free.tablespace_name
    WHERE ts_avail.tablespace_name IN ('NIS')
) a
LEFT JOIN (
    -- 子查询2：dba_data_files的文件数量/最大可扩展数据（别名b）
    SELECT
        df.tablespace_name,
        COUNT(*) AS file_count,
        SUM(df.bytes) AS total_allocated,
        -- 计算最大可扩展总字节数（处理自扩展+maxbytes=0的情况）
        SUM(CASE 
            WHEN df.autoextensible = 'YES' THEN NVL(NULLIF(df.maxbytes, 0), POWER(1024, 3) * 32)
            ELSE df.bytes 
        END) AS max_extend
    FROM dba_data_files df
    WHERE df.tablespace_name IN ('NIS')
    GROUP BY df.tablespace_name
) b ON a.tablespace_name = b.tablespace_name

ORDER BY a.tablespace_name DESC;
```

## 4.触发器

### 4.1 启动-删除触发器

```sql
-- 启用触发器
ALTER TRIGGER trg_met_ipm_execdrug_audit ENABLE;

-- 禁用触发器
ALTER TRIGGER trg_met_ipm_execdrug_audit DISABLE;

-- 重新编译触发器
ALTER TRIGGER trg_met_ipm_execdrug_audit COMPILE;

-- 删除触发器
DROP TRIGGER trg_met_ipm_execdrug_audit;
```
